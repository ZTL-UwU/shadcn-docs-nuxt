  {
    "name": "nuxt-app",

      "preview": "nuxt preview",

      "vue-tsc": "^2.2.0"
    }
  }
  ```

  ```json [tsconfig.json]
  {
    "extends": "./.nuxt/tsconfig.json"
  }
  ```

  ```ts [nuxt.config.ts]
  export default defineNuxtConfig({

    },

    css: ['~/assets/main.css']
  });
